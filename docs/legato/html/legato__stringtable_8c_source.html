184 leResult leStringTable_StringLookup(const leStringTable* table,
185  leStringInfo* info)
186 {
187  leStringTableEntry entry;
188  uint8_t* ptr;
189 
190  entry.length = 0;
191  entry.data = NULL;
192 
193  if(table == NULL || info == NULL)
194  return LE_FAILURE;
195 
196  info->offset = leStringTable_GetStringOffset(table,
197  info->stringIndex,
198  info->languageID);
199 
200  ptr = (uint8_t*)(table->stringTableData + info->offset);
201 
202  entry.length = ptr[0];
203  entry.length |= ptr[1] << 8;
204 
205  entry.data = (uint8_t*)table->stringTableData + info->offset + 2;
206 
207  info->data = entry.data;
208  info->dataSize = entry.length;
209  info->length = 0;
210 
211  return LE_SUCCESS;
