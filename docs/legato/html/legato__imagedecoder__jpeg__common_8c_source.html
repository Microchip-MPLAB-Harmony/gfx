77 /*******************************************************************************
78 Function: uint16_t JPEG_wReadWord(GFXU_ImageAsset *pfile)
79 
80 Precondition: None
81 
82 Overview: This function reads and returns a single word obtained as
83  Higher byte first followed by lower byte
84 
85 Input: Image file
86 
87 Output: One word
88 *******************************************************************************/
89 static uint16_t JPEG_wReadWord(JPEGDECODER *decoder)
90 {
91  uint8_t btemp;
92  uint16_t wData;
93 
94  decoder->readPtr(&btemp, sizeof(btemp), 1, decoder);
95  wData = btemp;
96  decoder->readPtr(&btemp, sizeof(btemp), 1, decoder);
97  wData = (wData << 8) | btemp;
98 
99  return wData;
100 }

122 /*******************************************************************************
123 Function: uint8_t JPEG_bReadHeader(JPEGDECODER *pJpegDecoder)
124 
125 Precondition: None
126 
127 Overview: This function reads the JPEG file header and
128  fills the data structure
129 
130 Input: JPEGDECODER
131 
132 Output: Error code - '0' means no error
133 *******************************************************************************/
134 uint8_t JPEG_bReadHeader(JPEGDECODER *pJpegDecoder)
135 {
136  leBool blSOSOver = 0;
137  while(pJpegDecoder->fileIndex * sizeof(uint8_t)
138  < pJpegDecoder->pImageFile->header.size)
139  {
140  uint8_t btemp, bcount, bsection;
141  uint16_t wSegLen, wOffset;
142 
143  if(blSOSOver == 1)
144  {
145  if(pJpegDecoder->bChannels == 1)
146  {
147  pJpegDecoder->bBlocksInOnePass = 1;
148  pJpegDecoder->bSubSampleType = JPEG_SAMPLE_1x1;
149  }
150  else if(pJpegDecoder->bChannels == 3)
151  {
152  if((pJpegDecoder->abChannelHSampFactor[0] == 1 && pJpegDecoder->abChannelVSampFactor[0] == 1) &&
153  (pJpegDecoder->abChannelHSampFactor[1] == 1 && pJpegDecoder->abChannelVSampFactor[1] == 1) &&
154  (pJpegDecoder->abChannelHSampFactor[2] == 1 && pJpegDecoder->abChannelVSampFactor[2] == 1))
155  {
156  pJpegDecoder->bBlocksInOnePass = 3;
157  pJpegDecoder->abChannelMap[0] = 0;
158  pJpegDecoder->abChannelMap[1] = 1;
159  pJpegDecoder->abChannelMap[2] = 2;
160  pJpegDecoder->bSubSampleType = JPEG_SAMPLE_1x1;
161  }
162  else if((pJpegDecoder->abChannelHSampFactor[0] == 1 && pJpegDecoder->abChannelVSampFactor[0] == 2) &&
163  (pJpegDecoder->abChannelHSampFactor[1] == 1 && pJpegDecoder->abChannelVSampFactor[1] == 1) &&
164  (pJpegDecoder->abChannelHSampFactor[2] == 1 && pJpegDecoder->abChannelVSampFactor[2] == 1))
165  {
166  pJpegDecoder->bBlocksInOnePass = 4;
167  pJpegDecoder->abChannelMap[0] = 0;
168  pJpegDecoder->abChannelMap[1] = 0;
169  pJpegDecoder->abChannelMap[2] = 1;
170  pJpegDecoder->abChannelMap[3] = 2;
171  pJpegDecoder->bSubSampleType = JPEG_SAMPLE_1x2;
172  }
173  else if((pJpegDecoder->abChannelHSampFactor[0] == 2 && pJpegDecoder->abChannelVSampFactor[0] == 1) &&
174  (pJpegDecoder->abChannelHSampFactor[1] == 1 && pJpegDecoder->abChannelVSampFactor[1] == 1) &&
175  (pJpegDecoder->abChannelHSampFactor[2] == 1 && pJpegDecoder->abChannelVSampFactor[2] == 1))
176  {
177  pJpegDecoder->bBlocksInOnePass = 4;
178  pJpegDecoder->abChannelMap[0] = 0;
179  pJpegDecoder->abChannelMap[1] = 0;
180  pJpegDecoder->abChannelMap[2] = 1;
181  pJpegDecoder->abChannelMap[3] = 2;
182  pJpegDecoder->bSubSampleType = JPEG_SAMPLE_2x1;
183  }
184  else if((pJpegDecoder->abChannelHSampFactor[0] == 2 && pJpegDecoder->abChannelVSampFactor[0] == 2) &&
185  (pJpegDecoder->abChannelHSampFactor[1] == 1 && pJpegDecoder->abChannelVSampFactor[1] == 1) &&
186  (pJpegDecoder->abChannelHSampFactor[2] == 1 && pJpegDecoder->abChannelVSampFactor[2] == 1))
187  {
188  pJpegDecoder->bBlocksInOnePass = 6;
189  pJpegDecoder->abChannelMap[0] = 0;
190  pJpegDecoder->abChannelMap[1] = 0;
191  pJpegDecoder->abChannelMap[2] = 0;
192  pJpegDecoder->abChannelMap[3] = 0;
193  pJpegDecoder->abChannelMap[4] = 1;
194  pJpegDecoder->abChannelMap[5] = 2;
195  pJpegDecoder->bSubSampleType = JPEG_SAMPLE_2x2;
196  }
197  else
198  {
199  JPEG_SendError(100);
200  }
201  }
202  return 0;
203  }
204 
205  pJpegDecoder->readPtr(&btemp, sizeof(btemp), 1, pJpegDecoder);
206 
207  if(btemp != 0xFF)
208  {
209  continue;
210  }
211 
212  pJpegDecoder->readPtr(&bsection, sizeof(bsection), 1, pJpegDecoder);
213  switch(bsection)
214  {
215  case SOI:
216  case TEM:
217  case EOI:
218  case RST0:
219  case RST1:
220  case RST2:
221  case RST3:
222  case RST4:
223  case RST5:
224  case RST6:
225  case RST7: break;
226 
227  case SOF0: /* Start of frame */
228  wSegLen = JPEG_wReadWord(pJpegDecoder);
229  if(wSegLen <= 8)
230  {
231  JPEG_SendError(100);
232  break;
233  }
234  pJpegDecoder->bDataBits = JPEG_bReadByte(pJpegDecoder);
235  pJpegDecoder->wHeight = JPEG_wReadWord(pJpegDecoder);
236  pJpegDecoder->wWidth = JPEG_wReadWord(pJpegDecoder);
237  pJpegDecoder->bChannels = JPEG_bReadByte(pJpegDecoder);
238  if(wSegLen < 8 + (pJpegDecoder->bChannels * 3))
239  {
240  JPEG_SendError(100);
241  break;
242  }
243  for(bcount = 0; bcount < pJpegDecoder->bChannels; bcount++)
244  {
245  pJpegDecoder->abChannelType[bcount] = JPEG_bReadByte(pJpegDecoder);
246  btemp = JPEG_bReadByte(pJpegDecoder);
247  pJpegDecoder->abChannelHSampFactor[bcount] = btemp >> 4;
248  pJpegDecoder->abChannelVSampFactor[bcount] = btemp & 0x0F;
249  pJpegDecoder->abChannelQuantTableMap[bcount] = JPEG_bReadByte(pJpegDecoder);
250  }
251  break;
252 
253  case APP0: /* Start of Application */
254  wSegLen = JPEG_wReadWord(pJpegDecoder);
255  if(wSegLen < 16)
256  {
257  JPEG_SendError(100);
258  break;
259  }
260  else
261  {
262  unsigned char buf[5];
263  pJpegDecoder->readPtr(buf, sizeof(buf[0]), 5, pJpegDecoder);
264  if(buf[0] == 'J' && buf[1] == 'F' && buf[2] == 'I' && buf[3] == 'F' && buf[4] == '\0')
265  {
266  pJpegDecoder->blJFIF = 1;
267  }
268  }
269  pJpegDecoder->bMajorRev = JPEG_bReadByte(pJpegDecoder);
270  pJpegDecoder->bMinorRev = JPEG_bReadByte(pJpegDecoder);
271  if(pJpegDecoder->bMajorRev != 1)
272  {
273  JPEG_SendError(100);
274  break;
275  }
276  /* Ignore other bytes in this segment */
277  break;
278 
279  case DRI: /* Start of Quantization table */
280  wSegLen = JPEG_wReadWord(pJpegDecoder);
281  if(wSegLen == 4)
282  {
283  pJpegDecoder->wRestartInterval = JPEG_wReadWord(pJpegDecoder);
284  }
285  break;
286 
287  case DQT: /* Start of Quantization table */
288  wSegLen = JPEG_wReadWord(pJpegDecoder);
289  if(wSegLen < 67)
290  {
291  JPEG_SendError(100);
292  break;
293  }
294 
295  do
296  {
297  uint8_t bQTableIndex, bCounter;
298 
299  btemp = JPEG_bReadByte(pJpegDecoder);
300  bQTableIndex = btemp & 0x0F;
301  pJpegDecoder->blQuantUses16bits |= (btemp >> 4);
302 
303  for(bCounter = 0; bCounter < 64; bCounter++)
304  {
305  uint8_t bData1, bData2 = 0;
306  if(pJpegDecoder->blQuantUses16bits == 0)
307  {
308  pJpegDecoder->readPtr(&bData1, sizeof(uint8_t), 1, pJpegDecoder);
309  pJpegDecoder->awQuantTable[bQTableIndex][abZigzag[bCounter]] = bData1;
310  }
311  else
312  {
313  pJpegDecoder->readPtr(&bData1, sizeof(uint8_t), 1, pJpegDecoder);
314  pJpegDecoder->readPtr(&bData2, sizeof(uint8_t), 1, pJpegDecoder);
315  pJpegDecoder->awQuantTable[bQTableIndex][abZigzag[bCounter]] = (((uint16_t)bData1) << 8) + (uint16_t)bData2;
316  }
317  }
318  wSegLen -= (pJpegDecoder->blQuantUses16bits == 0)? 65: 129; /* Table length + information byte */
319  } while(wSegLen >= 67);
320  break;
321 
322  case DHT: /* Start of Huffmann table */
323  wSegLen = JPEG_wReadWord(pJpegDecoder);
324  if(wSegLen < 19)
325  {
326  JPEG_SendError(100);
327  break;
328  }
329 
330  do
331  {
332  uint8_t bHTableIndex, bCounter;
333  uint16_t wNumOfSymbols;
334  uint8_t blIsAc;
335  uint8_t *pbLenTable, *pbSymTable;
336 
337  btemp = JPEG_bReadByte(pJpegDecoder);
338  bHTableIndex = btemp & 0x0F;
339  blIsAc = (btemp >> 4) & 0x01;
340 
341  if(blIsAc == 0)
342  {
343  pbLenTable = (uint8_t*)&pJpegDecoder->abHuffDcSymLen[bHTableIndex][0];
344  pbSymTable = (uint8_t*)&pJpegDecoder->abHuffDcSymbol[bHTableIndex][0];
345  }
346  else
347  {
348  pbLenTable = (uint8_t*)&pJpegDecoder->abHuffAcSymLen[bHTableIndex][0];
349  pbSymTable = (uint8_t*)&pJpegDecoder->abHuffAcSymbol[bHTableIndex][0];
350  }
351 
352  pJpegDecoder->readPtr(pbLenTable, sizeof(uint8_t), 16, pJpegDecoder);
353 
354  for(bCounter = 0, wNumOfSymbols = 0; bCounter < 16; bCounter++)
355  {
356  wNumOfSymbols += pbLenTable[bCounter];
357  }
358 
359  wSegLen -= 17; /* This table length + information byte */
360 
361  if(wSegLen < wNumOfSymbols || (blIsAc == 1 && wNumOfSymbols > 256) || (blIsAc == 0 && wNumOfSymbols > 16))
362  {
363  JPEG_SendError(100);
364  break;
365  }
366 
367  pJpegDecoder->readPtr(pbSymTable, sizeof(uint8_t), wNumOfSymbols, pJpegDecoder);
368  wSegLen -= wNumOfSymbols; /* This table length + information byte */
369  pJpegDecoder->bHuffTables++;
370  } while(wSegLen >= 19);
371  break;
372 
373  case SOS: /* Start of Scan parameters */
374  wSegLen = JPEG_wReadWord(pJpegDecoder);
375  if(wSegLen < 3)
376  {
377  JPEG_SendError(100);
378  break;
379  }
380 
381  btemp = JPEG_bReadByte(pJpegDecoder);
382  wOffset = wSegLen - (3 + (btemp << 1));
383 
384  if(pJpegDecoder->bChannels != btemp || wSegLen < 3 + (btemp << 1))
385  {
386  JPEG_SendError(100);
387  break;
388  }
389  else
390  {
391  uint8_t bCounter, bChannelId = 0xFF;
392 
393  for(bCounter = 0; bCounter < pJpegDecoder->bChannels; bCounter++)
394  {
395  uint8_t bindex;
396 
397  btemp = JPEG_bReadByte(pJpegDecoder);
398  for(bindex = 0; bindex < MAX_CHANNELS; bindex++)
399  {
400  if(pJpegDecoder->abChannelType[bindex] == btemp)
401  {
402  bChannelId = bindex;
403  break;
404  }
405  }
406 
407  if(bChannelId < MAX_CHANNELS)
408  {
409  btemp = JPEG_bReadByte(pJpegDecoder);
410  pJpegDecoder->abChannelHuffAcTableMap[bChannelId] = btemp & 0x0F;
411  pJpegDecoder->abChannelHuffDcTableMap[bChannelId] = btemp >> 4;
412  }
413  }
414  JPEG_Seek(pJpegDecoder, wOffset, 1);
415  }
416  blSOSOver = 1;
417  break;
418 
419  default: /* Any other segment with length */
420  wSegLen = JPEG_wReadWord(pJpegDecoder);
421  if(wSegLen < 2)
422  {
423  JPEG_SendError(100);
424  break;
425  }
426  JPEG_Seek(pJpegDecoder, wSegLen - 2, 1);
427  }
428  }
429  return (pJpegDecoder->blJFIF == 1) ? 0 : 1;
430 }

433 Function: uint8_t JPEG_bGenerateHuffmanTables(JPEGDECODER *pJpegDecoder)
434 
435 Precondition: None
436 
437 Overview: This function generated the table required for Huffman decoding
438  from the data read from the header
439 
440 Input: JPEGDECODER
441 
442 Output: Error code - '0' means no error
443 *******************************************************************************/
444 uint8_t JPEG_bGenerateHuffmanTables(JPEGDECODER *pJpegDecoder)
445 {
446  uint8_t bLength, bTable;
447 
448  for(bTable = 0; bTable < pJpegDecoder->bHuffTables / 2; bTable++)
449  {
450  pJpegDecoder->awHuffAcSymStart[bTable][0] = 0;
451  pJpegDecoder->awHuffDcSymStart[bTable][0] = 0;
452 
453  for(bLength = 1; bLength < 16; bLength++)
454  {
455  pJpegDecoder->awHuffAcSymStart[bTable][bLength] = (pJpegDecoder->awHuffAcSymStart[bTable][bLength - 1] + pJpegDecoder->abHuffAcSymLen[bTable][bLength - 1]) << 1;
456  pJpegDecoder->awHuffDcSymStart[bTable][bLength] = (pJpegDecoder->awHuffDcSymStart[bTable][bLength - 1] + pJpegDecoder->abHuffDcSymLen[bTable][bLength - 1]) << 1;
457  }
458  }
459  return 0;
460 }

462 /*******************************************************************************
463 Function: uint8_t JPEG_bGet1Bit(JPEGDECODER *pJpegDecoder)
464 
465 Precondition: None
466 
467 Overview: This function returns 1 bit as the lsb of the returned byte.
468  It automatically fills the buffer if it becomes empty.
469  It also converts 0xFF00 into 0.
470 
471 Input: JPEGDECODER
472 
473 Output: One bit
474 *******************************************************************************/
475 static uint8_t JPEG_bGet1Bit(JPEGDECODER *pJpegDecoder)
476 {
477  uint8_t bBit = 0;
478 
479  if(pJpegDecoder->wBufferIndex >= pJpegDecoder->wBufferLen)
480  {
481  pJpegDecoder->wBufferLen = pJpegDecoder->readPtr(&pJpegDecoder->abDataBuffer[0], sizeof(uint8_t), MAX_DATA_BUF_LEN, pJpegDecoder);
482  while(pJpegDecoder->abDataBuffer[pJpegDecoder->wBufferLen - 1] == 0xFF)
483  {
484  pJpegDecoder->wBufferLen--;
485  JPEG_Seek(pJpegDecoder, -1, 1);
486  }
487  pJpegDecoder->wBufferIndex = 0;
488  }
489 
490  while(pJpegDecoder->bBitsAvailable == 0) /* Be very careful to touch this part of code! You must know exactly what you are doing */
491  {
492  pJpegDecoder->bBitsAvailable = 16;
493  pJpegDecoder->wWorkBits = pJpegDecoder->abDataBuffer[pJpegDecoder->wBufferIndex++];
494  pJpegDecoder->wWorkBits = (pJpegDecoder->wWorkBits << 8) + pJpegDecoder->abDataBuffer[pJpegDecoder->wBufferIndex++];
495  if(pJpegDecoder->wBufferIndex > pJpegDecoder->wBufferLen) /* wBufferIndex need not be even because of the below condition */
496  {
497  pJpegDecoder->bBitsAvailable = 8;
498  }
499  else if((pJpegDecoder->wWorkBits & 0x00FF) == 0x00FF)
500  {
501  pJpegDecoder->bBitsAvailable = 8;
502  pJpegDecoder->wBufferIndex--;
503  }
504  else if(pJpegDecoder->wWorkBits >= 0xFF00)
505  {
506  if(pJpegDecoder->wWorkBits == 0xFF00)
507  {
508  pJpegDecoder->bBitsAvailable = 8;
509  }
510  }
511  }
512 
513  bBit = (pJpegDecoder->wWorkBits & 0x8000)? 0x01: 0;
514 
515  pJpegDecoder->wWorkBits <<= 1;
516  pJpegDecoder->bBitsAvailable--;
517 
518  return bBit;
519 }

521 /*******************************************************************************
522 Function: uint16_t JPEG_wGetBits(JPEGDECODER *pJpegDecoder, uint8_t bLen)
523 
524 Precondition: None
525 
526 Overview: This function returns bLen number of bits as the lsb of the
527  returned word and it automatically fills the buffer
528  if it becomes empty.
529 
530 Input: JPEGDECODER, Number of bits
531 
532 Output: Requested number of bits
533 *******************************************************************************/
534 static uint16_t JPEG_wGetBits(JPEGDECODER *pJpegDecoder, uint8_t bLen)
535 {
536  uint16_t wVal = 0;
537 
538  wVal = pJpegDecoder->bFirstBit = JPEG_bGet1Bit(pJpegDecoder);
539  bLen--;
540 
541  while(bLen)
542  {
543  wVal = (wVal << 1) + JPEG_bGet1Bit(pJpegDecoder);
544  bLen--;
545  }
546 
547  return wVal;
548 }

572 /*******************************************************************************
573 Function: int16_t JPEG_sGetBitsValue(JPEGDECODER *pJpegDecoder, uint8_t bLen)
574 
575 Precondition: None

577 Overview: Returns the signed value of the bLen bits of data
578 
579 Input: JPEGDECODER, Number of bits

599 /*******************************************************************************
600 Function: uint8_t JPEG_bGetNextHuffByte(JPEGDECODER *pJpegDecoder)
601 
602 Precondition: File pointer must point to the Huffman data
603 
604 Overview: Returns the Huffman decoded data
605 
606 Input: JPEGDECODER
607 
608 Output: Huffman decoded data
609 *******************************************************************************/
610 static uint8_t JPEG_bGetNextHuffByte(JPEGDECODER *pJpegDecoder)
611 {
612  uint8_t bBits, bSymbol = 0;
613  uint16_t wBitPattern = 0, wSymbolOffset = 0;
614 
615  for(bBits = 0; bBits < 16; bBits++)
616  {
617  uint8_t bSymbols;
618  uint16_t wDiff;
619 
620  wBitPattern = (wBitPattern << 1) + JPEG_bGet1Bit(pJpegDecoder);
621  bSymbols = pJpegDecoder->pbCurrentHuffSymLenTable[bBits];
622  if(bSymbols == 0)
623  {
624  continue;
625  }
626 
627  wDiff = wBitPattern - pJpegDecoder->pwCurrentHuffSymStartTable[bBits];
628  if(wDiff < bSymbols)
629  {
630  bSymbol = pJpegDecoder->pbCurrentHuffSymbolTable[wSymbolOffset + wDiff];
631  break;
632  }
633  wSymbolOffset += bSymbols;
634  }
635  return bSymbol;
636 }

639 /*******************************************************************************

651 uint8_t JPEG_bDecodeOneBlock(JPEGDECODER *pJpegDecoder)

663  if(wRestartWord < 0xFFD0 || wRestartWord > 0xFFD7)
664  {
665  JPEG_SendError(100);
666  }
667  for(bCounter = 0; bCounter < MAX_CHANNELS; bCounter++)
668  {
669  pJpegDecoder->asPrevDcValue[bCounter] = 0;
670  }
671  pJpegDecoder->wBlockNumber = 0;
672  }

675  {
676  pJpegDecoder->asOneBlock[bBlock][bCounter] = 0;
677  }
678 
679  pJpegDecoder->pwCurrentQuantTable = &pJpegDecoder->awQuantTable[pJpegDecoder->abChannelQuantTableMap[pJpegDecoder->abChannelMap[bBlock]]][0];
680 
681  /* Decode DC value */
682  bByteCount = 0;
683  pJpegDecoder->pbCurrentHuffSymLenTable = &pJpegDecoder->abHuffDcSymLen[pJpegDecoder->abChannelHuffDcTableMap[pJpegDecoder->abChannelMap[bBlock]]][0];
684  pJpegDecoder->pbCurrentHuffSymbolTable = &pJpegDecoder->abHuffDcSymbol[pJpegDecoder->abChannelHuffDcTableMap[pJpegDecoder->abChannelMap[bBlock]]][0];
685  pJpegDecoder->pwCurrentHuffSymStartTable = &pJpegDecoder->awHuffDcSymStart[pJpegDecoder->abChannelHuffDcTableMap[pJpegDecoder->abChannelMap[bBlock]]][0];
686  bHuffbyte = JPEG_bGetNextHuffByte(pJpegDecoder);
687  pJpegDecoder->asOneBlock[bBlock][0] = JPEG_sGetBitsValue(pJpegDecoder, bHuffbyte & 0x0F) + pJpegDecoder->asPrevDcValue[pJpegDecoder->abChannelMap[bBlock]];
688  pJpegDecoder->asPrevDcValue[pJpegDecoder->abChannelMap[bBlock]] = pJpegDecoder->asOneBlock[bBlock][0];
689 
690  /* Decode AC value */
691  bByteCount = 1;
692  pJpegDecoder->pbCurrentHuffSymLenTable = &pJpegDecoder->abHuffAcSymLen[pJpegDecoder->abChannelHuffAcTableMap[pJpegDecoder->abChannelMap[bBlock]]][0];
693  pJpegDecoder->pbCurrentHuffSymbolTable = &pJpegDecoder->abHuffAcSymbol[pJpegDecoder->abChannelHuffAcTableMap[pJpegDecoder->abChannelMap[bBlock]]][0];
694  pJpegDecoder->pwCurrentHuffSymStartTable = &pJpegDecoder->awHuffAcSymStart[pJpegDecoder->abChannelHuffAcTableMap[pJpegDecoder->abChannelMap[bBlock]]][0];
695  while(bByteCount < 64)
696  {
697  bHuffbyte = JPEG_bGetNextHuffByte(pJpegDecoder);
698  bByteCount += (bHuffbyte >> 4);
699  if(bHuffbyte == 0 /* EOB */)
700  {
701  break;
702  }
703  if(bByteCount > 63)
704  {
705  JPEG_SendError(100);
706  }
707  pJpegDecoder->asOneBlock[bBlock][abZigzag[bByteCount++]] = JPEG_sGetBitsValue(pJpegDecoder, bHuffbyte & 0x0F);
708  }
709  pJpegDecoder->wBlockNumber++;
710  jpeg_idct_islow(&pJpegDecoder->asOneBlock[bBlock][0],pJpegDecoder->pwCurrentQuantTable);
711  }
712 
713  return 0;
714 }
