273 static void snapRotation(leRadialMenuWidget* _this,
274  uint32_t ms)
275 {
276  if(_this->snapCounter < SNAP_TICK)
277  {
278  _this->snapCounter += ms;
279 
280  return;
281  }
282 
283  _this->snapCounter = 0;
284 
285  if(_this->rotationDegrees <= 1 && _this->rotationDegrees >= -1)
286  {
287  _this->snapState = SNAP_NONE;
288 
289  rotateMenu(_this, -_this->rotationDegrees);
290  }
291  else
292  {
293  rotateMenu(_this, -(_this->rotationDegrees / 2));
294  }
295 
296  _this->positionsInvalid = LE_TRUE;
297 
298  _this->fn->invalidate(_this);
299 }

301 static void updateRotation(leRadialMenuWidget* _this,
302  uint32_t ms)
303 {
304  uint32_t absMomentum = leAbsoluteValue(_this->momentum);
305  uint32_t sig, dec;
306  uint32_t fallOff;
307 
308  _this->rotationCounter += ms;
309 
310  _this->rotationCounter = 0;
311 
312  // if the user has a finger on the wheel, we should increase the friction
313  if (_this->touchPressed == LE_TRUE)
314  {
315  fallOff = _this->momentumFalloff * 4;
316  }
317  else
318  {
319  fallOff = _this->momentumFalloff;
320  }
321 
322  if (fallOff > 100)
323  {
324  fallOff = 100;
325  }
326 
327  lePercentOfDec(absMomentum,
328  100 - fallOff,
329  &sig,
330  &dec);
331 
332  if(_this->momentum < 0)
333  {
334  _this->momentum = -(int32_t)sig;
335 
336  rotateMenu(_this, -_this->momentum);
337  }
338  else
339  {
340  _this->momentum = sig;
341 
342  rotateMenu(_this, -_this->momentum);
343  }
344 
345  // below reasonable threshold, stop rotation
346  if(_this->momentum == 0 && _this->touchPressed == LE_FALSE && _this->rotationDegrees != 0)
347  {
348  _this->snapState = SNAP_PENDING;
349  _this->snapCounter = SNAP_DELAY;
350 
351  return;
352  }
353 }

355 static void update(leRadialMenuWidget* _this, uint32_t ms)
356 {
357  if(_this->ellipse.invalid == LE_TRUE)
358  {
359  _leRadialMenu_RebuildEllipse(_this);
360  }
361  else if(_this->positionsInvalid == LE_TRUE)
362  {
363  _leRadialMenu_RecalculateArrangement(_this);
364  }
365 
366  if(_this->momentum != 0)
367  {
368  updateRotation(_this, ms);
369  }
370 
371  if(_this->snapState == SNAP_PENDING)
372  {
373  _this->snapCounter -= ms;
374 
375  if(_this->snapCounter < 0)
376  {
377  _this->snapState = SNAP_INPROCESS;
378  _this->snapCounter = 0;
379  }
380  }
381  else if(_this->snapState == SNAP_INPROCESS)
382  {
383  snapRotation(_this, ms);
384  }
385 }

996 static float vectorLength(const struct Vector2* vec)
997 {
998  return leSqrt((vec->x * vec->x) + (vec->y * vec->y));
999 }

1001 static void projectVector(struct Vector2* vec, const struct Vector2* ref)
1002 {
1003  float f;
1004  float dot;
1005  float length;
1006 
1007  dot = (vec->x * ref->x) + (vec->y * ref->y);
1008 
1009  length = (ref->x * ref->x) + (ref->y * ref->y);
1010  f = dot / length;
1011 
1012  vec->x = ref->x * f;
1013  vec->y = ref->y * f;
1014 }

1016 static void rotateVector(struct Vector2* vec, float ang)
1017 {
1018  float cs, sn, x, y;
1019  float rad = ang * (3.14159f / 180.0f);
1020 
1021 
1022  cs = cosf(rad);
1023  sn = sinf(rad);
1024 
1025  x = vec->x * cs - vec->y * sn;
1026  y = vec->x * sn + vec->y * cs;
1027 
1028  vec->x = x;
1029  vec->y = y;
1030 }

1032 static void handleTouchMovedEvent(leRadialMenuWidget* _this,
1033  leWidgetEvent_TouchMove* evt)
1034 {
1035  leRect clipRect;
1036  struct Vector2 touchVector, rotationVector;
1037  float length, dot;
1038  int32_t rotationMagnitude;
1039 
1040  clipRect = _this->touchArea;
1041 
1042  leUtils_RectToScreenSpace((leWidget*)_this, &clipRect);
1043 
1044  rotationVector.x = 1.0f;
1045  rotationVector.y = 0.0f;
1046 
1047  rotateVector(&rotationVector, _this->ellipse.theta);
1048 
1049  touchVector.x = evt->x - evt->prevX;
1050  touchVector.y = evt->y - evt->prevY;
1051 
1052  dot = (touchVector.x * rotationVector.x) + (touchVector.y * rotationVector.y);
1053 
1054  projectVector(&touchVector, &rotationVector);
1055 
1056  length = vectorLength(&touchVector);
1057 
1058  if(length >= 0.75f && length < 1.0f)
1059  {
1060  length = 1.0f;
1061  }
1062 
1063  rotationMagnitude = (int32_t)length;
1064 
1065  if(dot < 0.0f)
1066  {
1067  rotationMagnitude *= -1;
1068  }
1069 
1070  _this->snapState = SNAP_NONE;
1071 
1072  if(rotationMagnitude > DEFAULT_DRAG_DELTA)
1073  {
1074  _this->momentum += rotationMagnitude;
1075 
1076  if(_this->momentum > MAX_MOMENTUM)
1077  {
1078  _this->momentum = MAX_MOMENTUM;
1079  }
1080  }
1081  else if(rotationMagnitude < -DEFAULT_DRAG_DELTA)
1082  {
1083  _this->momentum += rotationMagnitude;
1084 
1085  if(_this->momentum < -MAX_MOMENTUM)
1086  {
1087  _this->momentum = -MAX_MOMENTUM;
1088  }
1089  }
1090  else
1091  {
1092  rotateMenu(_this, -rotationMagnitude);
1093  }
1094 
1095  leWidgetEvent_Accept((leWidgetEvent*)evt, (leWidget*)_this);
1096 }
