93 void _leTextFieldWidget_GetCursorRect(const leTextFieldWidget* txt,
94  leRect* cursorRect)
95 {
96  leRect textRect;
97  leRect bounds;
98  leRect charRect;
99  const leString* str = NULL;
100 
101  txt->fn->localRect(txt, &bounds);
102 
103  if(txt->text.fn->length(&txt->text) > 0)
104  {
105  str = (leString*)&txt->text;
106  }
107  else if(txt->hintText != NULL && leGetEditWidget() != (void*)txt)
108  {
109  str = (leString*)txt->hintText;
110  }
111  else
112  {
113  textRect = leRect_Zero;
114  charRect = leRect_Zero;
115 
116  cursorRect->x = bounds.x + txt->editWidget.widget.margin.left;
117  }
118 
119  if(str != NULL)
120  {
121  str->fn->getRect(str, &textRect);
122 
123  leUtils_ArrangeRectangleRelative(&textRect,
124  leRect_Zero,
125  bounds,
126  txt->editWidget.widget.style.halign,
127  LE_VALIGN_MIDDLE,
128  0,
129  txt->editWidget.widget.margin.left,
130  txt->editWidget.widget.margin.top,
131  txt->editWidget.widget.margin.right,
132  txt->editWidget.widget.margin.bottom,
133  0);
134 
135  // start of the text
136  if(txt->cursorPos == 0)
137  {
138  cursorRect->x = textRect.x;
139  }
140  // end of the text
141  else if(txt->cursorPos >= str->fn->length(str))
142  {
143  cursorRect->x = textRect.x + textRect.width;
144  }
145  // middle of the text
146  else
147  {
148  str->fn->getCharRect(str, txt->cursorPos, &charRect);
149 
150  cursorRect->x = textRect.x + charRect.x;
151  }
152  }
153 
154 
155  cursorRect->y = bounds.y + txt->editWidget.widget.margin.top + 2;
156  cursorRect->width = 1;
157  cursorRect->height = bounds.height - txt->editWidget.widget.margin.top - txt->editWidget.widget.margin.bottom - 4;
158 
159  leUtils_RectToScreenSpace((leWidget*)txt, cursorRect);
160 }

167 static void nextState(leTextFieldWidget* txt)
168 {
169  switch(txt->editWidget.widget.status.drawState)
170  {
171  case NOT_STARTED:
172  {
173  paintState.alpha = 255;
174 
175 #if LE_ALPHA_BLENDING_ENABLED == 1
176  if(txt->fn->getCumulativeAlphaEnabled(txt) == LE_TRUE)
177  {
178  paintState.alpha = txt->fn->getCumulativeAlphaAmount(txt);
179  }
180 #endif
181 
182  if(txt->editWidget.widget.style.backgroundType != LE_WIDGET_BACKGROUND_NONE)
183  {
184  txt->editWidget.widget.status.drawState = DRAW_BACKGROUND;
185  txt->editWidget.widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
186 
187  return;
188  }
189  }
190  // fall through
191  case DRAW_BACKGROUND:
192  {
193  if(txt->text.fn->length(&txt->text) > 0 ||
194  (txt->hintText != NULL && leGetEditWidget() != (void*)txt))
195  {
196  txt->editWidget.widget.status.drawState = DRAW_STRING;
197  txt->editWidget.widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawString;
198 
199  return;
200  }
201  }
202  // fall through
203  case DRAW_STRING:
204  {
205  if(txt->cursorEnable == LE_TRUE && txt->cursorVisible == LE_TRUE)
206  {
207  txt->editWidget.widget.status.drawState = DRAW_CURSOR;
208  txt->editWidget.widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawCursor;
209 
210  return;
211  }
212  }
213  // fall through
214  case DRAW_CURSOR:
215  {
216  if(txt->editWidget.widget.style.borderType != LE_WIDGET_BORDER_NONE)
217  {
218  txt->editWidget.widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
219  txt->editWidget.widget.status.drawState = DRAW_BORDER;
220 
221  return;
222  }
223  }
224  // fall through
225  case DRAW_BORDER:
226  {
227  txt->editWidget.widget.status.drawState = DONE;
228  txt->editWidget.widget.drawFunc = NULL;
229  }
230  }
231 }

256 static void drawString(leTextFieldWidget* txt)
257 {
258  leRect textRect, drawRect;
259 
260  _leTextFieldWidget_GetTextRect(txt, &textRect, &drawRect);
261 
262  if(txt->text.fn->length(&txt->text) > 0)
263  {
264  txt->text.fn->_draw(&txt->text,
265  textRect.x,
266  textRect.y,
267  txt->editWidget.widget.style.halign,
268  leScheme_GetRenderColor(txt->editWidget.widget.scheme, LE_SCHM_TEXT),
269  paintState.alpha);
270  }
271  else if(txt->hintText != NULL && leGetEditWidget() != (void*)txt)
272  {
273  txt->hintText->fn->_draw(txt->hintText,
274  textRect.x,
275  textRect.y,
276  txt->editWidget.widget.style.halign,
277  leScheme_GetRenderColor(txt->editWidget.widget.scheme, LE_SCHM_TEXT_DISABLED),
278  paintState.alpha);
279  }
280 
281 #if LE_STREAMING_ENABLED == 1
282  if(leGetActiveStream() != NULL)
283  {
284  leGetActiveStream()->onDone = onStringStreamFinished;
285  leGetActiveStream()->userData = txt;
286 
287  txt->editWidget.widget.status.drawState = WAIT_STRING;
288 
289  return;
290  }
291 #endif
292 
293  nextState(txt);
294 }

326 void _leTextFieldWidget_Paint(leTextFieldWidget* txt)
327 {
328  if(txt->editWidget.widget.status.drawState == NOT_STARTED)
329  {
330  nextState(txt);
331  }
332 
333  while(txt->editWidget.widget.status.drawState != DONE)
334  {
335  txt->editWidget.widget.drawFunc((leWidget*)txt);
336 
337 #if LE_PREEMPTION_LEVEL == 2
338  break;
339 #endif
340 
341 #if LE_STREAMING_ENABLED == 1
342  if(txt->editWidget.widget.status.drawState == WAIT_STRING)
343  break;
344 #endif
345  }
346 }
