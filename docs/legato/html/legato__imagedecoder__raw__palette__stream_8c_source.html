87 static leResult exec_nonblocking(struct StreamPaletteStage* stage)
88 {
89  uint32_t addr;
90 
91  // no need to do anything if a data read is still pending
92  if(streamPaletteStage.state == SS_STALLED)
93  return LE_FAILURE;
94 
95  if(streamPaletteStage.state == SS_DONE)
96  {
97  streamPaletteStage.state = SS_READY;
98 
99  return LE_SUCCESS;
100  }
101 
102  streamPaletteStage.lookupIndex = streamPaletteStage.base.state->writeColor;
103 
104  addr = (uint32_t)streamPaletteStage.base.state->source->palette->header.address +
105  streamPaletteStage.lookupIndex *
106  stage->paletteSize;
107 
108  if(leStream_Read(&streamPaletteStage.stream,
109  addr,
110  stage->paletteSize,
111  (uint8_t*)&streamPaletteStage.base.state->writeColor,
112  indexDataReady) == LE_FAILURE)
113  {
114  return LE_FAILURE;
115  }
116 
117  // only stall out of the read is still pending
118  if(leStream_IsDataReady(&streamPaletteStage.stream) == LE_FALSE)
119  {
120  streamPaletteStage.state = SS_STALLED;
121 
122  return LE_FAILURE;
123  }
124 
125  advanceStage();
126 
127  if(streamPaletteStage.state == SS_DONE)
128  {
129  streamPaletteStage.state = SS_READY;
130 
131  return LE_SUCCESS;
132  }
133 
134  return LE_FAILURE;
135 }

137 static leResult exec_blocking(struct StreamPaletteStage* stage)
138 {
139  uint32_t addr;
140 
141  addr = (uint32_t)streamPaletteStage.base.state->source->palette->header.address +
142  streamPaletteStage.base.state->writeColor *
143  stage->paletteSize;
144 
145  while(leStream_Read(&streamPaletteStage.stream,
146  addr,
147  stage->paletteSize,
148  (uint8_t*)&streamPaletteStage.base.state->writeColor,
149  NULL) != LE_SUCCESS)
150  { }
151 
152  return LE_SUCCESS;
153 }

162 leResult _leRawImageDecoder_LookupStage_Stream(leRawDecodeState* state)
163 {
164  memset(&streamPaletteStage, 0, sizeof(streamPaletteStage));
165 
166  lePixelBufferCreate(LE_ASSET_DECODER_PALETTE_CACHE_SIZE / leColorInfoTable[state->source->palette->buffer.mode].size,
167  1,
168  state->source->palette->buffer.mode,
169  cache,
170  &streamPaletteStage.cacheBuffer);
171 
172  leStream_Init(&streamPaletteStage.stream,
173  (leStreamDescriptor*)state->source->palette,
174  LE_ASSET_DECODER_PALETTE_CACHE_SIZE,
175  leRawImageDecoderPaletteScratchBuffer,
176  NULL);
177 
178  if(leStream_Open(&streamPaletteStage.stream) == LE_FAILURE)
179  {
180  return LE_FAILURE;
181  }
182 
183  streamPaletteStage.paletteSize = leColorInfoTable[state->source->palette->buffer.mode].size;
184 
185  streamPaletteStage.base.state = state;
186  streamPaletteStage.base.cleanup = (void*)cleanup;
187 
188  if(leStream_IsBlocking(&streamPaletteStage.stream) == LE_TRUE)
189  {
190  streamPaletteStage.base.exec = (void*)exec_blocking;
191  }
192  else
193  {
194  streamPaletteStage.base.exec = (void*)exec_nonblocking;
195  }
196 
197  _leRawImageDecoder_InjectStage(state, (void*)&streamPaletteStage);
198 
199  return LE_SUCCESS;
200 }
