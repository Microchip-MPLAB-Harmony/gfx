44 static leResult stage_rotateNearestNeighborPreRead(struct RotateNearestNeighborPreReadStage* stage)
45 {
46  lePoint pnt;
47 
48  stage->base.state->readCount = 1;
49 
50  pnt.x = stage->base.state->colIterator;
51  pnt.y = stage->base.state->rowIterator;
52 
53  // transform into cartesian space
54  pnt.x -= stage->base.state->destRect.width / 2;
55  pnt.y -= stage->base.state->destRect.height / 2;
56  pnt.y *= -1;
57 
58  leRotatePoint(pnt,
59  lePoint_Zero,
60  stage->base.state->angle,
61  &pnt);
62 
63  // transform into source image space
64  pnt.y *= -1;
65  pnt.x += (stage->base.state->source->buffer.size.width / 2) - 1;
66  pnt.y += (stage->base.state->source->buffer.size.height / 2) - 1;
67 
68  /*if(stage->base.state->referenceX == 0 && stage->base.state->referenceY == 0)
69  {
70  printf("[%i, %i] : %i : [%i, %i]\n", stage->base.state->referenceX, stage->base.state->referenceY, stage->base.state->angle, pnt.x, pnt.y);
71  }*/
72 
73  if(pnt.x < 0 || pnt.x >= stage->base.state->source->buffer.size.width ||
74  pnt.y < 0 || pnt.y >= stage->base.state->source->buffer.size.height)
75  {
76  // this pixel position is invalid, reset the pipeline
77  stage->base.state->currentStage = -1;
78 
79  return LE_SUCCESS;
80  }
81 
82  stage->base.state->readOperation[0].x = pnt.x;
83  stage->base.state->readOperation[0].y = pnt.y;
84 
85  // calculate source index
86  stage->base.state->readOperation[0].bufferIndex = (stage->base.state->readOperation[0].x) +
87  ((stage->base.state->readOperation[0].y) *
88  stage->base.state->source->buffer.size.width);
89 
90  return LE_SUCCESS;
91 }

112 static leResult stage_bilinearPreRead(struct RotateBilinearPreReadStage* stage)
113 {
114  leRawDecodeState* state = stage->base.state;
115  lePoint readPoint, filterPoint;
116 
117  readPoint.x = stage->base.state->colIterator;
118  readPoint.y = stage->base.state->rowIterator;
119 
120  // transform into cartesian space
121  readPoint.x -= stage->base.state->destRect.width / 2;
122  readPoint.y -= stage->base.state->destRect.height / 2;
123  readPoint.y *= -1;
124 
125  leRotatePoint(readPoint,
126  lePoint_Zero,
127  stage->base.state->angle,
128  &readPoint);
129 
130  // transform into source image space
131  readPoint.y *= -1;
132  readPoint.x += (stage->base.state->source->buffer.size.width / 2) - 1;
133  readPoint.y += (stage->base.state->source->buffer.size.height / 2) - 1;
134 
135  if(readPoint.x < 0 || readPoint.x >= stage->base.state->source->buffer.size.width ||
136  readPoint.y < 0 || readPoint.y >= stage->base.state->source->buffer.size.height)
137  {
138  // this pixel position is invalid, reset the pipeline
139  stage->base.state->currentStage = -1;
140 
141  return LE_SUCCESS;
142  }
143 
144  // if we're right on the edge, do a nearest neighbor instead of a bilinear blend
145  if(readPoint.x == stage->base.state->source->buffer.size.width - 1 ||
146  readPoint.y == stage->base.state->source->buffer.size.height - 1)
147  {
148  state->readCount = 1;
149 
150  stage->base.state->readOperation[0].x = readPoint.x;
151  stage->base.state->readOperation[0].y = readPoint.y;
152 
153  state->readOperation[0].bufferIndex = readPoint.x + (readPoint.y * state->source->buffer.size.width);
154  }
155  else
156  {
157  state->readCount = 4;
158 
159  // first point
160  filterPoint = readPoint;
161 
162  stage->base.state->readOperation[0].x = filterPoint.x;
163  stage->base.state->readOperation[0].y = filterPoint.y;
164 
165  state->readOperation[0].bufferIndex = readPoint.x + (readPoint.y * state->source->buffer.size.width);
166 
167  // second
168  filterPoint = readPoint;
169 
170  if(filterPoint.x < stage->base.state->source->buffer.size.width - 1)
171  {
172  filterPoint.x += 1;
173  }
174 
175  stage->base.state->readOperation[1].x = filterPoint.x;
176  stage->base.state->readOperation[1].y = filterPoint.y;
177 
178  state->readOperation[1].bufferIndex = filterPoint.x + (filterPoint.y * state->source->buffer.size.width);
179 
180  // third
181  filterPoint = readPoint;
182 
183  if(filterPoint.y < stage->base.state->source->buffer.size.height - 1)
184  {
185  filterPoint.y += 1;
186  }
187 
188  stage->base.state->readOperation[2].x = filterPoint.x;
189  stage->base.state->readOperation[2].y = filterPoint.y;
190 
191  state->readOperation[2].bufferIndex = filterPoint.x + (filterPoint.y * state->source->buffer.size.width);
192 
193  // fourth
194  filterPoint = readPoint;
195 
196  if(filterPoint.x < stage->base.state->source->buffer.size.width - 1)
197  {
198  filterPoint.x += 1;
199  }
200 
201  if(filterPoint.y < stage->base.state->source->buffer.size.height - 1)
202  {
203  filterPoint.y += 1;
204  }
205 
206  stage->base.state->readOperation[3].x = filterPoint.x;
207  stage->base.state->readOperation[3].y = filterPoint.y;
208 
209  state->readOperation[3].bufferIndex = filterPoint.x + (filterPoint.y * state->source->buffer.size.width);
210  }
211 
212  return LE_SUCCESS;
213 }
