33 static leColor indexGet1bpp(const lePixelBuffer* const buffer,
34  const uint32_t idx)
35 {
36  uint32_t offs;
37  uint8_t byte;
38  leColor color;
39 
40  offs = (idx % 8);
41 
42  byte = *(((uint8_t*)buffer->pixels) + (idx >> 3));
43 
44  color = byte & (0x80 >> offs);
45 
46  color >>= 7 - offs;
47 
48  return color;
49 }

51 static leColor indexGet4bpp(const lePixelBuffer* const buffer,
52  const uint32_t idx)
53 {
54  uint32_t offs;
55  uint8_t byte;
56  leColor color;
57 
58  byte = *(((uint8_t*)buffer->pixels) + (idx >> 1));
59 
60  if(idx % 2 > 0)
61  {
62  offs = 0xF;
63  color = byte & offs;
64  }
65  else
66  {
67  offs = 0xF0;
68  color = (byte & offs) >> 0x4;
69  }
70 
71  return color;
72 }

74 static leColor indexGet(const lePixelBuffer* const buffer,
75  const uint32_t idx)
76 {
77  uint8_t* buf_ptr;
78  leColor color = 0;
79  uint32_t i;
80 
81  buf_ptr = ((uint8_t*)buffer->pixels) + (leColorInfoTable[buffer->mode].bpp >> 3) * idx;
82 
83  for(i = 0; i < leColorInfoTable[buffer->mode].bpp >> 3; i++)
84  ((uint8_t*)&color)[i] = buf_ptr[i];
85 
86  return color;
87 }

184 leResult lePixelBufferCreate(const int32_t width,
185  const int32_t height,
186  const leColorMode mode,
187  const void* const address,
188  lePixelBuffer* buffer)
189 {
190  if(buffer == NULL)
191  return LE_FAILURE;
192 
193  buffer->size.width = width;
194  buffer->size.height = height;
195  buffer->pixel_count = width * height;
196 
197  buffer->mode = mode;
198 
199  switch(mode)
200  {
201  case LE_COLOR_MODE_INDEX_1:
202  {
203  buffer->buffer_length = (width * height) / 8;
204 
205  if((width * height) % 8 > 0)
206  buffer->buffer_length += 1;
207 
208  break;
209  }
210  case LE_COLOR_MODE_INDEX_4:
211  {
212  buffer->buffer_length = (width * height) / 2;
213 
214  if((width * height) % 2 > 0)
215  buffer->buffer_length += 1;
216 
217  break;
218  }
219  case LE_COLOR_MODE_INDEX_8:
220  {
221  buffer->buffer_length = (width * height);
222 
223  break;
224  }
225  default:
226  {
227  buffer->buffer_length = width * height * ((leColorInfoTable[buffer->mode].bpp >> 3));
228  }
229  }
230 
231  buffer->pixels = (void*)address;
232 
233  return LE_SUCCESS;
234 }

236 leBuffer lePixelBufferOffsetGet(const lePixelBuffer* const buffer,
237  uint32_t x,
238  uint32_t y)
239 {
240  uint8_t* buf_ptr;
241 
242  if(buffer == NULL)
243  return NULL;
244 
245  buf_ptr = (uint8_t*)buffer->pixels;
246 
247  if(x * y >= buffer->pixel_count)
248  return NULL;
249 
250  return buf_ptr +
251  leColorInfoTable[buffer->mode].size *
252  (x + (y * buffer->size.width));
253 }

292 leResult lePixelBufferClipRect(const lePixelBuffer* const buffer,
293  const leRect* const rect,
294  leRect* result)
295 {
296  leRect buffer_rect;
297 
298  if(buffer == NULL || rect == NULL || result == NULL)
299  return LE_FAILURE;
300 
301  buffer_rect.x = 0;
302  buffer_rect.y = 0;
303  buffer_rect.width = buffer->size.width;
304  buffer_rect.height = buffer->size.height;
305 
306  // can't clip if rectangles do not overlap in any way
307  if(leRectIntersects(&buffer_rect, rect) == LE_FALSE)
308  return LE_FAILURE;
309 
310  // clip the rectangles
311  leRectClip(&buffer_rect, rect, result);
312 
313  return LE_SUCCESS;
314 }

394 leResult lePixelBufferAreaFill_Unsafe(const lePixelBuffer* const buffer,
395  uint32_t x,
396  uint32_t y,
397  uint32_t w,
398  uint32_t h,
399  leColor color)
400 {
401  int32_t row, row_max, col, col_max;
402  uint32_t drawX;
403  uint32_t rowSize;
404  void *srcIdx, *destIdx;
405 
406  // calculate minimums
407  row_max = h;
408  col_max = w;
409 
410  rowSize = col_max * leColorInfoTable[buffer->mode].size;
411 
412  // fill the first row
413  for(col = 0; col < col_max; col++)
414  {
415  drawX = x + col;
416 
417  // set dest pixel
418  lePixelBufferSet_Unsafe(buffer, drawX, y, color);
419  }
420 
421  srcIdx = lePixelBufferOffsetGet_Unsafe(buffer, x, y);
422 
423  // copy first row to subsequent rows
424  for(row = 1; row < row_max; row++)
425  {
426  destIdx = lePixelBufferOffsetGet_Unsafe(buffer, x, y + row);
427  //idx = lePixelBufferOffsetGet_Unsafe(buffer, x, y + row);
428 
429  //idx = (uint8_t*)(((uint32_t)buffer->pixels) + ((y + row) * rowSize));
430 
431  memcpy(destIdx, srcIdx, rowSize);
432  }
433 
434  return LE_SUCCESS;
435 }

437 leResult lePixelBufferCopy(lePixelBuffer* dest,
438  uint32_t x,
439  uint32_t y,
440  const lePixelBuffer* src,
441  const leRect* srcRect)
442 {
443  int32_t row, col;
444  leRect bufferRect, clippedRect;
445  leColor color;
446 
447  bufferRect.x = 0;
448  bufferRect.y = 0;
449  bufferRect.width = src->size.width;
450  bufferRect.height = src->size.height;
451 
452  leRectClip(&bufferRect, srcRect, &clippedRect);
453 
454  for(row = 0; row < clippedRect.height; ++row)
455  {
456  for(col = 0; col < clippedRect.width; ++col)
457  {
458  color = lePixelBufferGet(src, srcRect->x + col, srcRect->y + row);
459 
460  color = leColorConvert(src->mode, dest->mode, color);
461 
462  lePixelBufferSet_Unsafe(dest, x + col, y + row, color);
463  }
464  }
465 
466  return LE_SUCCESS;
