65 void drawGlyphRow1bpp(const uint8_t* data,
66  //int32_t width,
67  int32_t x,
68  int32_t y,
69  int32_t colStart,
70  int32_t colEnd,
71  leColor clr,
72  uint32_t a)
73 {
74  int32_t i;
75  uint32_t offs;
76  uint8_t val;
77 
78  a = leClampi(0, 255, a);
79 
80  if(a < 255)
81  {
82  for(i = colStart; i < colEnd; i++)
83  {
84  offs = (i % 8);
85 
86  val = data[i >> 3] & (0x80 >> offs);
87 
88  if(val > 0)
89  {
90  leRenderer_BlendPixel(x + i, y, clr, a);
91  }
92  }
93  }
94  else
95  {
96  for(i = colStart; i < colEnd; i++)
97  {
98  offs = (i % 8);
99 
100  val = data[i >> 3] & (0x80 >> offs);
101 
102  if(val > 0)
103  {
104  leRenderer_PutPixel(x + i, y, clr);
105  }
106  }
107  }
108 }

110 void drawGlyphRow8bpp(const uint8_t* data,
111  //int32_t width,
112  int32_t x,
113  int32_t y,
114  int32_t colStart,
115  int32_t colEnd,
116  leColor clr,
117  uint32_t a)
118 {
119  int32_t i;
120  uint32_t valPercent;
121  uint32_t aPercent;
122 
123  a = leClampi(0, 255, a);
124 
125  if(a < 255)
126  {
127  aPercent = lePercentWholeRounded(a, 255);
128 
129  for(i = colStart; i < colEnd; i++)
130  {
131  if(data[i] > 0)
132  {
133  valPercent = lePercentWholeRounded(data[i], 255);
134  valPercent = lePercentOf(aPercent, valPercent);
135  valPercent = lePercentOf(valPercent, 255);
136 
137  leRenderer_BlendPixel(x + i, y, clr, valPercent);
138  }
139  }
140  }
141  else
142  {
143  for(i = colStart; i < colEnd; i++)
144  {
145  if(data[i] > 0)
146  {
147  leRenderer_BlendPixel(x + i, y, clr, data[i]);
148  }
149  }
150  }
151 }

171 leResult leFont_GetGlyphInfo(const leFont* fnt,
172  uint32_t codepoint,
173  leFontGlyph* glyph)
174 {
175  uint32_t idx = 0;
176  uint32_t glyphCount;
177  leFontGlyph* glyphTable;
178  leRasterFont* rasFont = (leRasterFont*)fnt;
179 
180  LE_ASSERT(fnt != NULL && rasFont->glyphTable != NULL);
181 
182  glyphCount = ((uint32_t*)rasFont->glyphTable)[0];
183  glyphTable = (leFontGlyph*)(rasFont->glyphTable + sizeof(uint32_t));
184 
185  // attempt to find the glyph in the range lookup tables
186  for(idx = 0; idx < glyphCount; idx++)
187  {
188 
189  /* O1 linear search, faster searches could be implemented but would require more
190  memory */
191  if(codepoint == glyphTable[idx].codePoint)
192  {
193  *glyph = glyphTable[idx];
194 
195  return LE_SUCCESS;
196  }
197  }
198 
199  if(codepoint == 0x20)
200  {
201  // return space glyph
202  glyph->codePoint = 0x20;
203  glyph->width = glyph->width;
204  glyph->height = 0;
205  glyph->advance = glyph->width;
206  glyph->bearingX = 0;
207  glyph->bearingY = 0;
208  glyph->dataOffset = 0;
209  glyph->dataRowWidth = 0;
210  }
211  else
212  {
213  // return unknown glyph
214  glyph->codePoint = -1;
215  glyph->width = rasFont->height >> 1;
216  glyph->height = rasFont->height;
217  glyph->advance = glyph->width;
218  glyph->bearingX = 0;
219  glyph->bearingY = rasFont->height;
220  glyph->dataOffset = 0;
221  glyph->dataRowWidth = 0;
222  }
223 
224  return LE_FAILURE;
225 }

290 leResult leFont_DrawGlyphData(const leFont* fnt,
291  const leFontGlyph* glyph,
292  const uint8_t* data,
293  int32_t x,
294  int32_t y,
295  leColor clr,
296  uint32_t a)
297 {
298  int32_t row;
299  leRect glyphRect;
300  leRect clippedGlyph;
301  int32_t colStart, colEnd;
302  leRasterFont* rasFnt = (leRasterFont*)fnt;
303 
304  //x += glyph->bearingX;
305  //y += (fnt->baseline - glyph->bearingY);
306 
307  glyphRect.x = x;
308  glyphRect.y = y;
309  glyphRect.width = glyph->width;
310  glyphRect.height = glyph->height;
311 
312  if(leRenderer_CullDrawRect(&glyphRect) == LE_TRUE)
313  return LE_SUCCESS;
314 
315  leRenderer_ClipDrawRect(&glyphRect, &clippedGlyph);
316 
317  if(x >= clippedGlyph.x)
318  {
319  colStart = 0;
320  }
321  else
322  {
323  colStart = clippedGlyph.x - x;
324  }
325 
326  if(x + (int32_t)glyph->width <= clippedGlyph.x + clippedGlyph.width)
327  {
328  colEnd = glyph->width;
329  }
330  else
331  {
332  colEnd = glyph->width + ((clippedGlyph.x + clippedGlyph.width) - (x + (int32_t)glyph->width));
333  }
334 
335  for(row = clippedGlyph.y - glyphRect.y; row < (clippedGlyph.y - glyphRect.y) + (clippedGlyph.height); row++)
336  {
337  drawGlyphRowFn[rasFnt->bpp](data + (row * glyph->dataRowWidth),
338  x,
339  y + row,
340  colStart,
341  colEnd,
342  clr,
343  a);
344  }
345 
346  return LE_SUCCESS;
347 }

349 leResult leFont_DrawGlyph(const leFont* fnt,
350  const leFontGlyph* glyph,
351  int32_t x,
352  int32_t y,
353  leColor clr,
354  uint32_t a)
355 {
356  uint8_t* addr;
357 
358  if(fnt == NULL || glyph == NULL)
359  return LE_FAILURE;
360 
361  // attempt to find the glyph raster data
362  // if no data found then draw empty rectangle
363  if(glyph->codePoint == LE_UNKNOWN_GLYPH)
364  {
365  leFont_DrawUnknownGlyph(x, y, glyph, clr, a);
366 
367  return LE_SUCCESS;
368  }
369  else
370  {
371  if(fnt->header.location == LE_STREAM_LOCATION_ID_INTERNAL)
372  {
373  addr = ((uint8_t*)fnt->header.address);
374  addr += glyph->dataOffset;
375 
376  return leFont_DrawGlyphData(fnt, glyph, addr, x, y, clr, a);
377  }
378  }
379 
380  return LE_FAILURE;
381 }
