107 static void _recalculateSliceAngles(const lePieChartWidget* _this)
108 {
109  lePieChartPie* pie;
110  uint32_t lastPercent = 0;
111  uint32_t totalValue = 0;
112 
113  uint32_t itr;
114 
115  if(_this->pieArray.size == 0)
116  return;
117 
118  // calculate total chart value
119  for(itr = 0; itr < _this->pieArray.size; itr++)
120  {
121  pie = (lePieChartPie*)_this->pieArray.values[itr];
122 
123  totalValue += pie->value;
124  }
125 
126  // calculate slice percentages
127  for(itr = 0; itr < _this->pieArray.size; itr++)
128  {
129  pie = (lePieChartPie*)_this->pieArray.values[itr];
130 
131  pie->percentOffset = lastPercent;
132  pie->percentValue = lePercentWholeRounded(pie->value, totalValue);
133 
134  lastPercent = pie->percentOffset + pie->percentValue;
135 
136  if(itr == _this->pieArray.size - 1)
137  {
138  if(pie->percentOffset + pie->percentValue != 100)
139  {
140  pie->percentValue = 100 - pie->percentOffset;
141  }
142  }
143  }
144 
145  // calculate slice angles
146  for(itr = 0; itr < _this->pieArray.size; itr++)
147  {
148  pie = (lePieChartPie*)_this->pieArray.values[itr];
149 
150  pie->startAngle = leDegreesFromPercent(pie->percentOffset,
151  _this->centerAngle,
152  _this->startAngle);
153 
154  pie->spanAngle = leDegreesFromPercent(pie->percentValue,
155  _this->centerAngle,
156  0);
157 
158  if(itr == _this->pieArray.size - 1)
159  {
160  if(pie->startAngle + pie->spanAngle != _this->startAngle + _this->centerAngle)
161  {
162  pie->spanAngle = _this->startAngle + _this->centerAngle - pie->startAngle;
163  }
164  }
165  }
166 }

502 static int32_t _getPieIndexAtPoint(lePieChartWidget* _this,
503  lePoint pt)
504 {
505  lePoint centerPt;
506  float rad, mag, dot, det;
507  int32_t deg;
508  float xf, yf;
509  uint32_t itr, midAngle;
510  lePieChartPie* pie;
511  lePoint offsetPnt, modPt;
512 
513  // find the slice that the point angle intersects
514  for(itr = 0; itr < _this->pieArray.size; itr++)
515  {
516  pie = (lePieChartPie*)_this->pieArray.values[itr];
517 
518  modPt = pt;
519 
520  // transform the vector
521  centerPt.x = _this->widget.rect.width / 2;
522  centerPt.y = _this->widget.rect.height / 2;
523 
524  if (pie->offset != 0)
525  {
526  // find center angle of pie
527  midAngle = pie->startAngle + (pie->spanAngle / 2);
528 
529  // find points of
530  lePolarToXY(pie->offset, midAngle , &offsetPnt);
531  offsetPnt.y *= -1;
532 
533  centerPt.x += offsetPnt.x;
534  centerPt.y += offsetPnt.y;
535  }
536 
537  modPt.x -= centerPt.x;
538  modPt.y -= centerPt.y;
539  modPt.y *= -1;
540 
541  // normalize the vector
542  mag = leSqrt((modPt.x * modPt.x) + (modPt.y * modPt.y));
543 
544  xf = (float)modPt.x / mag;
545  yf = (float)modPt.y / mag;
546 
547  // calculate the angle
548  dot = (xf * 1.0f) + (yf * 0.0f);
549  det = (xf * 0.0f) - (yf * 1.0f);
550 
551  rad = -atan2f(det, dot);
552 
553  deg = (int32_t)(rad * 57.295827909f);
554 
555  if(deg < 0)
556  {
557  deg += 360;
558  }
559 
560  if(deg >= (int32_t)pie->startAngle && deg <= (int32_t)pie->startAngle + (int32_t)pie->spanAngle)
561  {
562  // ensure the point is inside the slice area
563  if(mag < pie->radius)
564  return itr;
565  }
566  }
567 
568  return -1;
569 }

571 static void handleTouchDownEvent(lePieChartWidget* _this,
572  leWidgetEvent_TouchDown* evt)
573 {
574  lePoint pnt;
575  int32_t index;
576 
577  pnt.x = evt->x;
578  pnt.y = evt->y;
579 
580  // adjust point relative to widget local
581  leUtils_PointScreenToLocalSpace((leWidget*)_this, &pnt);
582 
583  index = _getPieIndexAtPoint(_this, pnt);
584 
585  if(index != -1)
586  {
587  leWidgetEvent_Accept((leWidgetEvent*)evt, (leWidget*)_this);
588 
589  if(_this->pressedCallback != NULL)
590  {
591  _this->pressedCallback(_this, index);
592  }
593  }
